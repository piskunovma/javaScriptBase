Дан код: var a = 1, b = 1, c, d; c = ++a; alert(c); // 2 d = b++; alert(d); // 1 c = (2+ ++a); alert(c); // 5 d = (2+ b++); alert(d); // 4 alert(a); // 3 alert(b); // 3 Почему код даёт именно такие результаты? 1) ++ перед а - префиксная форма (сначала увеличиваем, потом присваиваем) 2) ++ после b - постфиксная форма (сначала присваеваем, потом увеличиваем) 3) 2 + ++а = 5 (2 + 2 + 1 = 2 + 3 = 5) 4) 2 + b++ = 4 (2 + 2 = 4 (b = 2)) после действия b стало равно 3 5) a = 3 (1 + 1 + 1) 6) b = 3 (1 + 1 + 1)
